  <script>
    // Demo items
    const list = document.getElementById('list');
    for (let i = 1; i <= 60; i++) {
      const el = document.createElement('div');
      el.className = 'item';
      el.innerHTML = `
        <div class="dot"></div>
        <div class="meta">
          <b>Item ${i}</b>
          <span>Detalle / descripción del item ${i}</span>
        </div>
      `;
      list.appendChild(el);
    }

    const scroller = document.getElementById('scroller');
    const header = document.getElementById('header');
    const hero = header.querySelector('.hero');
    const bottom = header.querySelector('.row-bottom');

    document.getElementById('toTop').addEventListener('click', () => {
      scroller.scrollTo({ top: 0, behavior: 'smooth' });
    });

    const supportsSDA =
      CSS.supports('scroll-timeline: --t block') &&
      CSS.supports('animation-timeline: --t') &&
      CSS.supports('animation-range: 0px 1px');

    const chip = document.getElementById('chip');
    chip.textContent = supportsSDA ? 'SDA: ON' : 'SDA: OFF (fallback)';

    const clamp01 = (x) => Math.max(0, Math.min(1, x));
    const getVarPx = (name, fallback) => {
      const v = getComputedStyle(document.documentElement).getPropertyValue(name).trim();
      const n = parseFloat(v);
      return Number.isFinite(n) ? n : fallback;
    };

    const expanded = getVarPx('--expanded', 180);
    const collapsed = getVarPx('--collapsed', 64);
    const range = getVarPx('--collapse-range', 140);

    const toggleBtn = document.getElementById('toggleHeader');
    let manualExpanded = false;

    function collapsedByScroll() {
      const t = clamp01(scroller.scrollTop / range);
      return t >= 0.99;
    }

    function updateToggleVisibility() {
      const isCollapsed = collapsedByScroll();
      const show = manualExpanded || isCollapsed;

      toggleBtn.hidden = !show;
      toggleBtn.textContent = manualExpanded ? 'Contraer' : 'Expandir';
      toggleBtn.setAttribute('aria-expanded', manualExpanded ? 'true' : 'false');
    }

    function applyFallbackFromScroll() {
      if (manualExpanded) return;

      const t = clamp01(scroller.scrollTop / range);
      const h = expanded + (collapsed - expanded) * t;

      header.style.height = `${h}px`;

      const o = 1 - t;
      hero.style.opacity = `${o}`;
      hero.style.transform = `translateY(${(-6 * t)}px)`;
      bottom.style.opacity = `${o}`;
      bottom.style.transform = `translateY(${(-6 * t)}px)`;
    }

    // ===== Compensación anti-brinco (anchor-based) =====
    function pickAnchorForCompensation() {
      const rect = scroller.getBoundingClientRect();
      const x = rect.left + Math.min(40, rect.width / 2);

      // Elegimos un Y dentro de la lista (debajo del header actual)
      const yOffset = manualExpanded ? (expanded + 28) : (collapsed + 28);
      const y = Math.min(rect.bottom - 20, rect.top + yOffset);

      const el = document.elementFromPoint(x, y);
      const anchor = el?.closest?.('.item') || null;
      if (!anchor) return null;

      return { el: anchor, top: anchor.getBoundingClientRect().top };
    }

    function compensateAfterLayout(anchor) {
      if (!anchor?.el) return;
      const newTop = anchor.el.getBoundingClientRect().top;
      const delta = newTop - anchor.top;
      if (Math.abs(delta) > 0.5) scroller.scrollTop += delta;
    }

    function setManualExpanded(next) {
      if (manualExpanded === next) return;
      if (!manualExpanded && !collapsedByScroll()) return;

      const anchor = pickAnchorForCompensation();
      const prevBehavior = scroller.style.scrollBehavior;
      scroller.style.scrollBehavior = 'auto';

      manualExpanded = next;
      document.body.classList.toggle('manual-expanded', manualExpanded);

      if (!supportsSDA) {
        if (manualExpanded) {
          header.style.height = `${expanded}px`;
          hero.style.opacity = `1`;
          hero.style.transform = `translateY(0)`;
          bottom.style.opacity = `1`;
          bottom.style.transform = `translateY(0)`;
        }
      }

      requestAnimationFrame(() => {
        if (!supportsSDA && !manualExpanded) applyFallbackFromScroll();

        compensateAfterLayout(anchor);
        requestAnimationFrame(() => {
          compensateAfterLayout(anchor);
          scroller.style.scrollBehavior = prevBehavior;
          updateToggleVisibility();
        });
      });
    }

    toggleBtn.addEventListener('click', () => setManualExpanded(!manualExpanded));

    function onScroll() {
      if (!supportsSDA) applyFallbackFromScroll();

      if (manualExpanded && scroller.scrollTop < range * 0.6) {
        setManualExpanded(false);
        return;
      }

      updateToggleVisibility();
    }

    scroller.addEventListener('scroll', onScroll, { passive: true });

    if (!supportsSDA) document.body.classList.add('no-sda');
    if (!supportsSDA) applyFallbackFromScroll();
    updateToggleVisibility();
  </script>
